def onStep():
    if GameCover.visible==True:
        droplet.visible=True
        droplet.toFront()
        droplet.centerY+=droplet.dy
        waterJug.visible=True
        waterJug.toFront()
        waterJug2.visible=True
        waterJug2.toFront()
        water.visible=True
        water.toFront()
        if waterJug.contains(droplet.centerX,droplet.centerY) and droplet.top>258:
            droplet.visible=False
            score.value+=1
            droplet.bottom=0
            droplet.centerX=randrange(15,385)
            droplet.dy*=1.1
        elif droplet.top>330:
            droplet.bottom=0
            droplet.centerX=randrange(15,385)
            droplet.dy*=1.1
            droplet.visible=True
            misses.value+=1
        ###This is where the custom function is called, at the end
           of the game:
        if score.value==20:
            endGame('YOU WIN!',"(press 'r' to play again!)")
        if misses.value==10:
            endGame('GAME OVER!',"(press 'r' to try again)")
