#This is the code for our book shelf. The parameters 
are its y position and its visibility. Conditional statements 
are included within the function so that as the y value of 
the shelf increases, the color becomes darker.
def drawShelf(y,visible):
    shelf=Rect(10,y,380,80,fill='burlyWood', border='saddleBrown', borderWidth=4,visible=visible)
    if shelf.centerY<160:
        shelf.fill='burlyWood'
    elif shelf.centerY<240:
        shelf.fill='sienna'
    else:
        shelf.fill='maroon'

Project 4: Trash Dash: Cleaning up the Ocean

This game supports the UN Global Goal of Life under Water. The player must first answer three educational questions concerning the cleanliness of the ocean, and then move on to play a game in which they must control a shrimp to avoid trash from hitting it.

Concepts Learned:

  • if, elif, else statements
  • key events
  • custom shape properties
  • randomizers

  #This function draws a correct answer with a prompt that changes 
   depending on which question is answered in the game.
  def drawCorrect():
    if (Label5.value=='Question 3:'):
        GoOn2.visible=True
        GoOn2.toFront()
    else:
        GoOn1.visible=True
        GoOn1.toFront()
    Rect(130,150,150,70,fill='lightBlue')
    CorrectReply.visible=True
    CorrectReply.toFront()
  #This is where the correct answer with the prompts are called. 
   Based on the question number, the prompt changes.
  ###Also, the following code uses 'hits'. For this function, it means that 
     if where the mouse is pressed hits the specified answer, then the 'Correct!'
     message will appear.
  def onMousePress(mouseX,mouseY):
    if (Ans2.hits(mouseX,mouseY)and(Label5.value=='Question 1:')):
        print(100)
        drawCorrect()
    if (Ans3.hits(mouseX,mouseY))and(Label5.value=='Question 2:'):
        print(200)
        drawCorrect()
    if (Ans1.hits(mouseX,mouseY))and(Label5.value=='Question 3:'):
        print(200)
        drawCorrect()
  #This is a key event! This code makes the game so that holding down 
   the 'left' or 'right' key will move the shrimp left or right!
  def onKeyHold(key):
    if 'left' in key:
        ShrimpGroup.centerX-=10
    if 'right' in key:
        ShrimpGroup.centerX+=10
  def onStep():
    if app.fall==True:
        CigaretteGroup.toFront()
        if CigaretteGroup.top<400:
            CigaretteGroup.centerY+=15
        elif CigaretteGroup.top>=400:
            CigaretteGroup.bottom=50
            CigaretteGroup.centerX=randrange(10,390)
        #In this part of the code, if the shrimp contains the center of the 
         cigarette butt, the app will stop running, ending the game.
        if ShrimpGroup.contains(CigaretteGroup.centerX,CigaretteGroup.centerY):
            app.stop()
            GameGroup.visible=True
            GameGroup.toFront()
